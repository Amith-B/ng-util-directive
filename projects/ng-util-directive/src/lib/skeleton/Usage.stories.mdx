# Skeleton Usage

`import { NgSkeletonModule } from 'ng-util-directive';`

<br />

## FullSize

`fullSize` shape will take the complete width and height of the dom content on which the directive is used

<Story id="directives-ngskeleton--basic-full-size" />

```html
<h1
  style="min-width: 120px; min-height: 40px"
  *ngSkeleton="ngSkeleton; data: { shape: 'fullSize' }; hideOnLoading: true"
>
  Heading
</h1>

<p
  style="min-width: 250px; min-height: 20px"
  *ngSkeleton="true; data: { shape: 'fullSize' }; hideOnLoading: true"
>
  First paragraph with some content
</p>

<p
  style="min-width: 250px; min-height: 20px"
  *ngSkeleton="true; data: { shape: 'fullSize' }; hideOnLoading: true"
>
  Second paragraph with some content
</p>
```

## Circle

`circle` shape fits into the size specified from size value

<Story id="directives-ngskeleton--circle" />

```html
<img
  (load)="loading = false"
  *ngSkeleton="loading; data: {shape: 'circle', size: '100px' }; hideOnLoading: true"
  style="width: 100px; height: 100px; border-radius: 50%;"
  src="avatar.svg"
/>
```

## Square

`square` shape fits into the size specified from size value

<Story id="directives-ngskeleton--square" />

```html
<img
  (load)="loading = false"
  *ngSkeleton="
    loading;
    data: {shape: 'square', size: '100px' };
    hideOnLoading: true"
  style="width: 200px; height: 200px; border-radius: 20px;"
  src="art.jpg"
/>
```

## Rectangle

`rectangle` shape fits into the size specified from size value

<Story id="directives-ngskeleton--rectangle" />

```html
<img
  (load)="loading = false"
  *ngSkeleton="
    loading;
    data: {shape: 'rectangle', width: 400px, height: 200px};
    hideOnLoading: true"
  style="width: 400px; height: 200px; border-radius: 20px;"
  src="art.jpg"
/>
```

# Directive Props

```typescript tsx
DirectiveProps {
  ngSkeleton: boolean;
  data: ShinyLoaderData | unknown;
  component: ThisType<unknown>;
  hideOnLoading: boolean;
}
```

# ShinyLoaderData type

```typescript tsx
export type ShinyLoaderData = (FullSize | Circle | Square | Rectangle) & {
  loaderColor1?: string;
  loaderColor2?: string;
};

export type FullSize = {
  shape: "fullSize";
  borderRadius?: string;
};

export type Circle = {
  shape: "circle";
  size: string;
};

export type Square = {
  shape: "square";
  size: string;
  borderRadius?: string;
};

export type Rectangle = {
  shape: "rectangle";
  borderRadius?: string;
  width: string;
  height: string;
};
```
